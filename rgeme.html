// Function to handle the purchase button click event async function purchaseDomain() { try { const domainName = document.getElementById("domainName").value; const tld = document.getElementById("tld").value; const duration = parseInt(document.getElementById("duration").value); const price = await contract.renewalPrice(); const tx = await contract.registerDomain(domainName, tld, 0, false, duration, { value: price.mul(duration) }); await tx.wait(); alert("Domain registered!"); } catch (error) { console.error("Error registering domain:", error); showStatus("roleStatus", "Error registering domain", "error"); } } // Function to handle the renewal button click event async function renewDomain() { try { const tokenId = parseInt(document.getElementById("tokenId").value); const duration = parseInt(document.getElementById("duration").value); const price = await contract.renewalPrice(); const tx = await contract.renewDomain(tokenId, duration, { value: price.mul(duration) }); await tx.wait(); alert("Domain renewed!"); } catch (error) { if (error.code === 4001) { alert("Transaction cancelled by user."); } else { console.error("Error renewing domain:", error); showStatus("roleStatus", "Error renewing domain", "error"); } } }
